 &= \mean{X^2(t_1)}+\mean{X^2(t_2)}-2\mean{X(t_1)X(t_2)}\\
           & = 2\mean{X^2(t_1)}-2\mean{X(t_1)X(t_1+2)}\\
           & = 2R_X(0)-2R_X(2)

        \end{align}

            \end{split}
            \label{process/1/auto}
            \end{align}
        From     \eqref{process/1},  \eqref{process/1/var} and  \eqref{process/1/auto},
