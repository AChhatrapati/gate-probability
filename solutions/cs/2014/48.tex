\begin{align}
\Pr(A+B) &= 1\\
\Pr(A) + \Pr(B) &= 1 \\
\Pr(A)\Pr(B) &= \Pr(A) (1 - \Pr(A))\\
&= \Pr(A) - (\Pr(A))^2\\
&= \dfrac{1}{4} - \left( \Pr(A) - \dfrac{1}{2} \right)^2 \\
&\leq \dfrac{1}{4}\\
\Pr(A) = \Pr(B) = \dfrac{1}{2} &\Rightarrow \Pr(A)\Pr(B) = \dfrac{1}{4}\\ \therefore \text{max}(\Pr(A)\Pr(B)) &= \dfrac{1}{4}
\end{align}
