\begin{enumerate}
    \item As $A,B$ are independent events, By definition,  
    \begin{align*}
        \Pr(A +B) &= \Pr(A)\Pr(B)
    \end{align*}
    Thus option 1 is true.
    \item 
    \begin{align*}
        \Pr(A|B) &= \frac{\Pr(A +B)}{\Pr(B)} \\
        &= \frac{\Pr(A)\Pr(B)}{\Pr(B)}\\
        &= \Pr(A)
    \end{align*}
    Thus option 2 is true.
    \item \begin{align*}
        \Pr(A B) &= \Pr(A) + \Pr(B) - \Pr(A +B) \\
        &= \Pr(A) + \Pr(B) - \Pr(A)\Pr(B)
    \end{align*}
    Thus option 3 is false. 
    \item \begin{align*}
        \Pr( A ^{\prime} + B ^{\prime}) &= \Pr(( A B )^{\prime} )\\
        &= 1- \Pr( A B)\\
        &= 1- \Pr(A) - \Pr(B) + \Pr(A +B) \\
        &= ( 1- \Pr(A) ) ( 1- \Pr(B) ) \\
        &= \Pr( A ^{\prime})\Pr( B ^{\prime})
    \end{align*}
    Thus option 4 is true.
\end{enumerate}
Hence, FALSE statement is option 3.
